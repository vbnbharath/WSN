95 // Strobe commands

97 #define SFSTXON 0x31 // Enable/calibrate freq synthesizer
98 #define SXOFF 0x32 // Turn off crystal oscillator.
99 #define SCAL 0x33 // Calibrate freq synthesizer & disable
100 #define SRX 0x34 // Enable RX.
101 #define STX 0x35 // Enable TX.
102 #define SIDLE 0x36 // Exit RX / TX
103 #define SAFC 0x37 // AFC adjustment of freq synthesizer
104 #define SWOR 0x38 // Start automatic RX polling sequence
105 #define SPWD 0x39 // Enter pwr down mode when CSn goes hi
106 #define SFRX 0x3A // Flush the RX FIFO buffer.
107 #define SFTX 0x3B // Flush the TX FIFO buffer.
108 #define SWORRST 0x3C // Reset real time clock.
109 #define SNOP 0x3D // No operation.
